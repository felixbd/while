** usage

*** compile

#+begin_src shell
stack build
#+end_src


*** run

#+begin_src shell
stack run
#+end_src


*** test

#+begin_src shell
stack test
#+end_src


*** linting

#+begin_src shell
stack exec hlint src/*.hs app/*.hs test/*.hs
#+end_src
